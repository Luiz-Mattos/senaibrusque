segue abaixo as tabelas 

Usuários.

CREATE TABLE `usuario` ( 
`id_user` TINYINT( 3 ) NOT NULL ,
`usuario` VARCHAR( 20 ) NOT NULL ,
`senha` VARCHAR( 32 ) NOT NULL ,
`nivel` TINYINT( 3 ) NOT NULL ,
PRIMARY KEY ( `id_user` ) ,
UNIQUE ( 
`usuario` 
)
) ENGINE = innodb;




NOTICIAS.

CREATE TABLE `noticias` (
  `inicial` varchar(200) NOT NULL,
  `descricao` varchar(100) NOT NULL,
  `titulo` varchar(200) NOT NULL,
  `noticia` varchar(3000) NOT NULL,
  `email_autor` varchar(30) NOT NULL,
  `direcao` varchar(10) NOT NULL,
  `id_noticia` tinyint(5) NOT NULL auto_increment,
  PRIMARY KEY  (`id_noticia`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 AUTO_INCREMENT=3 ;
